### **4.2 Swipe-to-Close Gesture (JavaScript)**

```tsx
// src/components/Header.tsx

import React, { useState, useRef } from 'react';

const Header: React.FC = () => {
  const [mobileOpen, setMobileOpen] = useState(false);
  const touchStartX = useRef(0);
  const touchEndX = useRef(0);

  const handleTouchStart = (e: React.TouchEvent) => {
    touchStartX.current = e.targetTouches[0].clientX;
  };

  const handleTouchMove = (e: React.TouchEvent) => {
    touchEndX.current = e.targetTouches[0].clientX;
  };

  const handleTouchEnd = () => {
    const swipeDistance = touchStartX.current - touchEndX.current;

    // –°–≤–∞–π–ø –≤–ª—ñ–≤–æ –Ω–∞ 75px ‚Äî –∑–∞–∫—Ä–∏—Ç–∏ –º–µ–Ω—é
    if (swipeDistance > 75) {
      setMobileOpen(false);
    }

    // –°–≤–∞–π–ø –≤–ø—Ä–∞–≤–æ –Ω–∞ 75px ‚Äî —Ç–∞–∫–æ–∂ –∑–∞–∫—Ä–∏—Ç–∏ (—è–∫—â–æ –º–µ–Ω—é –≤—ñ–¥–∫—Ä–∏—Ç–æ)
    if (swipeDistance < -75 && mobileOpen) {
      setMobileOpen(false);
    }
  };

  return (
    <header>
      {/* ... */}

      {mobileOpen && (
        <div
          className="mobile-menu-overlay"
          onTouchStart={handleTouchStart}
          onTouchMove={handleTouchMove}
          onTouchEnd={handleTouchEnd}
          onClick={() => setMobileOpen(false)} // –ö–ª—ñ–∫ –ø–æ–∑–∞ –º–µ–Ω—é —Ç–∞–∫–æ–∂ –∑–∞–∫—Ä–∏–≤–∞—î
        >
          <nav className="mobile-menu-content" onClick={(e) => e.stopPropagation()}>
            {/* Nav items */}
          </nav>
        </div>
      )}
    </header>
  );
};
```
